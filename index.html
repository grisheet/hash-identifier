Quick Start

# Clone the repo
git clone https://github.com/grisheet/hash-identifier.git
cd hash-identifier

# Identify a hash
python -m hash_identifier identify 5d41402abc4b2a76b9719d911017c592

# Crack a hash with a wordlist
python -m hash_identifier crack 5d41402abc4b2a76b9719d911017c592 --wordlist wordlist.txt

Python API

from hash_identifier import identify_hash, crack_hash

# Identify hash type
results = identify_hash("5d41402abc4b2a76b9719d911017c592")
for r in results:
    print(r.hash_name, r.confidence)

# Crack a hash
result = crack_hash("5d41402abc4b2a76b9719d911017c592", "wordlist.txt")
if result:
    print("Cracked:", result.plaintext)
